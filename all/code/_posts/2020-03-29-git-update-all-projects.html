用shell脚本，批量更新git仓库。
脚本会循环两层文件夹，寻找.git文件夹 来断定是否是git仓库，从而更新。 

#!/bin/sh
for dir in $(ls -d */)
do
  cd $dir
  echo "into $dir"
  if [ -d ".git" ]; then
     git pull
  else
     for dir in $(ls -d */)
	 do
	   cd $dir
	   echo "into $dir"
	   if [ -d ".git" ]; then
		  git pull
	   fi
	   cd ..
	 done
  fi
  cd ..
done
保存为git_pull.sh文件
在git bash中执行“./git_pull.sh”即可。
